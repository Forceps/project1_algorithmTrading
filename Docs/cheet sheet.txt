1. Linux 기본 조작 명령
  ls                             # 현재 디렉토리의파일과 폴더 출력
  cd [경로]                      # 입력한 경로로 이동(상대 경로)  # 예를 들어 cd ..   -> 상위 디렉토리로 이동
  sudo -i                        # 관리자 권한 가지기
  mkdir [폴더 이름]               # 디렉토리 생성
  rm [파일 이름]                  # 파일 삭제
  rm -rf [폴더 이름]              # 폴더 삭제
  cp [현재 경로] [복사 위치]       #복사
  du [대상 이름]                  # 대상의 용량 확인(kb 단위)
